struct DynamicHeightView: View {
    let text1: String
    let text2: String

    var body: some View {
        VStack {
            Text(text1)
                .frame(maxWidth: .infinity, alignment: .leading)
                .fixedSize(horizontal: false, vertical: true)

            Text(text2)
                .frame(maxWidth: .infinity, alignment: .leading)
                .fixedSize(horizontal: false, vertical: true)
        }
        .padding()
    }
}

struct ContentView: View {
    var body: some View {
        DynamicHeightView(
            text1: "This is the first text. It can be long and should wrap properly.",
            text2: "This is the second text. It will also expand dynamically based on its content."
        )
        .padding()
    }
}
