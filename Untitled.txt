import SwiftUI

struct ContentView: View {
    let items: [(String, Int)] = [
        ("Apple", 10),
        ("Banana", 5),
        ("Orange", 8)
    ]

    var body: some View {
        List(0..<items.count, id: \.self) { index in
            HStack {
                Text(items[index].0) // Access tuple's first value (name)
                    .font(.headline)
                Spacer()
                Text("\(items[index].1)") // Access tuple's second value (count)
                    .font(.subheadline)
                    .foregroundColor(.gray)
            }
            .padding()
        }
    }
}

struct ContentView_Previews: PreviewProvider {
    static var previews: some View {
        ContentView()
    }
}
