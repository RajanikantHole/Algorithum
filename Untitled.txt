private func setupBindings() {
    // Binding is done for Account card and Transaction list
    SisMasked
        .sink { [weak self] newValue in
            guard let self else { return }
            
            self.transactionsListViewModel.isMasked = newValue
            self.cardAccountViewModel.isMasked = newValue
            self.accountDetailsViewModel.isMasked = newValue
            
            if newValue {
                self.cardAccountViewModel.cardInfoModel.accountAmountText = 
                    "\(AccountConstants.ComponentStrings.masked) \(AccountConstants.ComponentStrings.currency)"
                self.headerIcon = [AccountConstants.ComponentStrings.hideIconTitle]
            } else {
                self.cardAccountViewModel.cardInfoModel.accountAmountText = self.amountValue ?? ""
                self.headerIcon = [AccountConstants.ComponentStrings.iconTitle]
            }
        }
}
