class User: Codable {
    let name: String
    let email: String
    let age: String
    let city: String
    let isActive: String
    let joinDate: String

    init(name: String, email: String, age: String, city: String, isActive: String, joinDate: String) {
        self.name = name
        self.email = email
        self.age = age
        self.city = city
        self.isActive = isActive
        self.joinDate = joinDate
    }
}
