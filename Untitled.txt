@MainActor
func getNoTransactionScreenContentHeight() -> CGFloat {
    let screenHeight = UIScreen.main.bounds.height
    let additionalSpacing: CGFloat = 180  // Extra padding or reserved space
    let usedHeight = noTransactionScreenContentHeight + 
                     DesignConstants.TabBarComponentDimensions.tabBarMaxHeight + 
                     additionalSpacing
    
    return max(screenHeight - usedHeight, 350)
}
