struct ContentView: View {
    @State private var isLarge = false

    var body: some View {
        Text("Tap Me")
            .padding()
            .background(Color.green)
            .cornerRadius(10)
            .if(isLarge) { view in
                view.frame(width: 200, height: 100)
            }
            .onTapGesture {
                isLarge.toggle()
            }
    }
}

extension View {
    @ViewBuilder
    func `if`<Content: View>(_ condition: Bool, transform: (Self) -> Content) -> some View {
        if condition {
            transform(self)
        } else {
            self
        }
    }
}
