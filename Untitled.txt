init() {
    SisMasked
        .sink { [weak self] isMasked in
            guard let self else { return }

            for (index, transaction) in self.transactions.enumerated() {
                let currency = transaction.currencyCode ?? ""

                if isMasked {
                    transaction.amount = "\(AccountConstants.ComponentStrings.masked) \(currency)"
                    transaction.balance = "\(AccountConstants.ComponentStrings.balanceLabel) \(AccountConstants.ComponentStrings.masked) \(currency)"
                } else if index < self.amountArray.count, index < self.balanceArray.count {
                    transaction.amount = "\(self.amountArray[index]) \(currency)"
                    transaction.balance = "\(AccountConstants.ComponentStrings.balanceLabel) \(self.balanceArray[index]) \(currency)"
                }
            }
        }
}
