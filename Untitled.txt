init() {
    SisMasked
        .sink { [weak self] isMasked in
            guard let self else { return }
            
            guard let listItemContract = self.listItemContract else { return }
            
            if isMasked {
                let maskedValue = "\(AccountConstants.ComponentStrings.masked) \(self.nonMakedlistItemContract?.currencyCode ?? "")"
                
                listItemContract.currentBalance = maskedValue
                listItemContract.lienAmount = maskedValue
                listItemContract.availableBalance = maskedValue
            } else {
                listItemContract.currentBalance = self.nonMakedlistItemContract?.currentBalance ?? ""
                listItemContract.lienAmount = self.nonMakedlistItemContract?.lienAmount ?? ""
                listItemContract.availableBalance = self.nonMakedlistItemContract?.availableBalance ?? ""
            }
        }
}
