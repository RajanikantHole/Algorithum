do {
    let transactions = try await useCase.executeFetchTransactions(accountNumber: someAccountNumber)
    
    self.transactions = transactions.transactionList.map { object in
        TransactionItemDataSource(
            type: object.transactionDescription,
            amount: "\(object.amount)",
            date: "\(object.dateTime)",
            balance: "\(object.runningBalance)",
            referenceNumber: object.refNumber,
            isCredit: true,
            currencyCode: object.currencyCode
        )
    }
} catch {
    print("Failed to fetch transactions: \(error)")
}
