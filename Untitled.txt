init() {
    SisMasked
        .sink { [weak self] isMasked in
            guard let self else { return }
            
            for index in 0..<self.transactions.count {
                if isMasked {
                    self.transactions[index].amount = 
                        "\(AccountConstants.ComponentStrings.masked) \(self.transactions[index].currencyCode ?? "")"
                    
                    self.transactions[index].balance = 
                        "\(AccountConstants.ComponentStrings.balanceLabel) \(AccountConstants.ComponentStrings.masked) \(self.transactions[index].currencyCode ?? "")"
                } else if !self.transactions.isEmpty {
                    self.transactions[index].amount = 
                        "\(self.amountArray[index]) \(self.transactions[index].currencyCode ?? "")"
                    
                    self.transactions[index].balance = 
                        "\(AccountConstants.ComponentStrings.balanceLabel) \(self.balanceArray[index]) \(self.transactions[index].currencyCode ?? "")"
                }
            }
        }
}
